# ======================================================
# Configuração Apache - Sistema Viabix
# ======================================================

# Ativar rewrite engine
RewriteEngine On

# ======================================================
# Regras de segurança
# ======================================================

# Bloquear acesso a arquivos sensíveis
<FilesMatch "\.(htaccess|htpasswd|ini|log|sh|sql|bak|config)$">
    Order Allow,Deny
    Deny from all
</FilesMatch>

# Bloquear listagem de diretórios
Options -Indexes

# ======================================================
# Proteção contra ataques comuns
# ======================================================

# Prevenir clickjacking
Header always append X-Frame-Options SAMEORIGIN

# Ativar proteção XSS no navegador
Header set X-XSS-Protection "1; mode=block"

# Prevenir MIME-type sniffing
Header set X-Content-Type-Options "nosniff"

# ======================================================
# Compressão GZIP
# ======================================================

<IfModule mod_deflate.c>
    AddOutputFilterByType DEFLATE text/html text/plain text/xml text/css text/javascript
    AddOutputFilterByType DEFLATE application/javascript application/x-javascript application/json
    AddOutputFilterByType DEFLATE application/xml application/xhtml+xml application/rss+xml
    AddOutputFilterByType DEFLATE application/x-font-ttf application/x-font-opentype image/svg+xml
</IfModule>

# ======================================================
# Cache de arquivos estáticos
# ======================================================

<IfModule mod_expires.c>
    ExpiresActive On
    
    # Cache por 1 ano para arquivos estáticos
    ExpiresByType image/jpg "access plus 1 year"
    ExpiresByType image/jpeg "access plus 1 year"
    ExpiresByType image/gif "access plus 1 year"
    ExpiresByType image/png "access plus 1 year"
    ExpiresByType image/svg+xml "access plus 1 year"
    ExpiresByType text/css "access plus 1 month"
    ExpiresByType application/javascript "access plus 1 month"
    ExpiresByType application/x-javascript "access plus 1 month"
    ExpiresByType text/javascript "access plus 1 month"
</IfModule>

# ======================================================
# Redirecionamentos e regras de reescrita
# ======================================================

# Forçar HTTPS (descomentar em produção)
# RewriteCond %{HTTPS} off
# RewriteRule ^(.*)$ https://%{HTTP_HOST}/$1 [R=301,L]

# Redirecionar para index.html para rotas do frontend (SPA)
RewriteCond %{REQUEST_FILENAME} !-f
RewriteCond %{REQUEST_FILENAME} !-d
RewriteRule ^ index.html [QSA,L]

# ======================================================
# Configurações de PHP (via .htaccess)
# ======================================================

<IfModule mod_php7.c>
    # Tamanho máximo de upload
    php_value upload_max_filesize 50M
    php_value post_max_size 50M
    php_value max_execution_time 300
    php_value max_input_time 300
    php_value memory_limit 256M
    
    # Configurações de sessão
    php_value session.gc_maxlifetime 28800
    php_value session.cookie_lifetime 0
    php_value session.use_strict_mode 1
    
    # Exibição de erros (desativar em produção)
    php_flag display_errors on
    php_flag display_startup_errors on
    php_flag log_errors on
</IfModule>

Header set Referrer-Policy "strict-origin-when-cross-origin"

# Permissions Policy
Header set Permissions-Policy "geolocation=(), microphone=(), camera=()"
